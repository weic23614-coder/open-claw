三Excel 解析注意事项
⚠️ 重要：出库 Excel 使用 inlineStr 或 sharedStrings 格式，不能直接用 pandas.read_excel()，必须用 zipfile + ElementTree 解析。 
import zipfile
import xml.etree.ElementTree as ET

def parse_xlsx(filepath):
    with zipfile.ZipFile(filepath) as z:
        # 读取 sharedStrings（可能不存在）
        shared = []
        if 'xl/sharedStrings.xml' in z.namelist():
            root = ET.fromstring(z.read('xl/sharedStrings.xml'))
            ns = {'x': 'http://schemas.openxmlformats.org/spreadsheetml/2006/main'}
            for si in root.findall('x:si', ns):
                shared.append(''.join(t.text or '' for t in si.findall('.//x:t', ns)))

        # 解析 sheet
        root = ET.fromstring(z.read('xl/worksheets/sheet1.xml'))
        ns = {'x': 'http://schemas.openxmlformats.org/spreadsheetml/2006/main'}

        def get_col(ref):
            col = ''.join(c for c in ref if c.isalpha())
            idx = 0
            for c in col: idx = idx * 26 + (ord(c) - ord('A') + 1)
            return idx - 1

        def get_val(cell):
            t = cell.get('t', '')
            v = cell.find('x:v', ns)
            if t == 's': return shared[int(v.text)] if v is not None else ''
            if t == 'inlineStr':
                it = cell.find('.//x:t', ns)
                return it.text if it is not None else ''
            return v.text if v is not None else ''

        data = []
        for row in root.findall('.//x:row', ns):
            cells = row.findall('x:c', ns)
            row_dict = {get_col(c.get('r','')): get_val(c) for c in cells}
            data.append(row_dict)
        return data  # data[0] 是表头行
